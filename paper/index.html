下面这段代码是一个使用Python的Scrapy框架编写的爬虫， 用于从中国人力资源和社会保障部网站上抓取特定的PDF公开数据文件， 并保存到本地目录作为数据支撑来用。 通过下载 人社部统计公报.RAR 来了解我的论文数据集。 

# 使用Python Scrapy 框架进行爬取中国人社部公开数据作为数据支撑
import scrapy
from scrapy import Request
import os
import time


class MohrssSpider(scrapy.Spider):
    name = "mohrss"
    allowed_domains = ["www.mohrss.gov.cn"]
    start_urls = [
        "https://www.mohrss.gov.cn/SYrlzyhshbzb/zwgk/szrs/tjsj/"
    ]

    # 初始化一个计数器用于生成唯一的文件名
    pdf_counter = 0

    def parse(self, response):
        # 获取列表页的链接
        list_pages = response.xpath('//a[@title[contains(.,"人力资源和社会保障主要统计数据")]]/@href').getall()

        # 构造完整的URL路径并请求
        base_url = "https://www.mohrss.gov.cn"
        for list_page in list_pages:
            full_list_page_url = response.urljoin(list_page)
            yield Request(full_list_page_url, callback=self.parse_list_page)

    def parse_list_page(self, response):
        # 从列表页中获取详情页的链接
        detail_pages = response.xpath('//div[@class="TRS_Editor"]/a/@href').getall()

        # 构造完整的URL路径并请求
        base_url = "https://www.mohrss.gov.cn"
        for detail_page in detail_pages:
            full_detail_page_url = response.urljoin(detail_page)
            yield Request(full_detail_page_url, callback=self.save_pdf)



    def save_pdf(self, response):
        # 检查响应是否为PDF文件
        content_type = response.headers.get(b'Content-Type', b'').decode('utf-8')
        if 'application/pdf' in content_type:
            # 定义保存路径
            base_path = 'output/'
            os.makedirs(base_path, exist_ok=True)  # 创建输出目录

            # 使用计数器生成唯一的文件名
            file_name = f"{int(time.time() * 1000)}_{self.pdf_counter}.pdf"
            self.pdf_counter += 1  # 更新计数器

            path = os.path.join(base_path, file_name)

            with open(path, 'wb') as f:
                f.write(response.body)
            print(f"PDF 文件已保存至 {path}")
        else:
            print("响应不是有效的PDF文件")


# 运行爬虫
if __name__ == '__main__':
    from scrapy.crawler import CrawlerProcess

    process = CrawlerProcess({
        'USER_AGENT': 'Mozilla/4.0 (compatible; MSIE 7.0; Windows NT 5.1)'
    })

    process.crawl(MohrssSpider)
    process.start()
